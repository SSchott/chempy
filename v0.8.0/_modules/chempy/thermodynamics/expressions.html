[docs]    def active_conc_prod(self, variables, backend=math, equilibrium=None):
        result = None
        for exp_factor, stoichs in [(1, equilibrium.prod), (-1, equilibrium.reac)]:
            for k, v in stoichs.items():
                if result is None:
                    result = variables[k]**(exp_factor*v)
                else:
                    result *= variables[k]**(exp_factor*v)
        return result
